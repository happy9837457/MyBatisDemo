   (	"USER_ID" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"USER_NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"USER_PWD" VARCHAR2(20 BYTE) NOT NULL ENABLE,
